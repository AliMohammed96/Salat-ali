    async function getPrayerTimes() {

      try {

        const pos = await new Promise((resolve, reject) => {

          navigator.geolocation.getCurrentPosition(resolve, reject);

        });

        const lat = pos.coords.latitude;

        const lon = pos.coords.longitude;

        const res = await fetch(`https://api.aladhan.com/v1/timings?latitude=${lat}&longitude=${lon}&method=5`);

        const data = await res.json();

        const timings = data.data.timings;

        const container = document.getElementById("prayer-times");

        container.innerHTML = "";

        for (const [name, time] of Object.entries(timings)) {

          if (["Fajr","Dhuhr","Asr","Maghrib","Isha"].includes(name)) {

            const div = document.createElement("div");

            div.className = "prayer-time";

            div.innerHTML = `<strong>${translateName(name)}</strong>: ${time}`;

            container.appendChild(div);

          }

        }

        setInterval(() => checkAdhan(timings), 30000);

      } catch (err) {

        document.getElementById("prayer-times").innerText = "فشل في تحديد الموقع.";

      }

    }

    function translateName(name) {

      const names = {

        Fajr: "الفجر",

        Dhuhr: "الظهر",

        Asr: "العصر",

        Maghrib: "المغرب",

        Isha: "العشاء"

      };

      return names[name] || name;

    }

    function checkAdhan(timings) {

      const now = new Date();

      const current = now.getHours().toString().padStart(2, '0') + ":" + now.getMinutes().toString().padStart(2, '0');

      for (const time of Object.values(timings)) {

        if (time === current) {

          document.getElementById("adhan").play();

          break;

        }

      }

    }

    getPrayerTimes();
